<script>
const DICTEES = [
    "Les fleurs que j'ai cueillies ce matin sont déjà fanées.",
    "Sidi et son frère sont partis au village pour voir leur grand-père.",
    "Il faut que tu fasses attention aux accords du participe passé.",
    "Les oiseaux se sont envolés dès qu'ils ont entendu ce grand bruit.",
    "Où vas-tu ? Je vais à la boutique pour acheter du pain et du lait.",
    "Ses amis sont venus, mais il ne leur a pas parlé longtemps.",
    "C'est une magnifique journée, le soleil brille sur Dakar.",
    "Les élèves sérieux finissent toujours par réussir leurs examens.",
    "Regarde ces enfants qui jouent avec leurs nouveaux joujoux.",
    "Nous mangerons ensemble quand vous aurez terminé votre travail.",
    "Si j'avais su, je serais venu plus tôt pour vous aider.",
    "Le maître a posé sa règle sur la table de la classe.",
    "Ils croient que nous sommes déjà arrivés à destination.",
    "Mame Anna semble fatiguée à cause de la longue marche.",
    "Les efforts fournis portent enfin leurs fruits.",
    "Elle s'est lavé les mains avant de passer à table.",
    "Quels que soient vos problèmes, gardez toujours espoir.",
    "La pluie est tombée toute la nuit sur les toits de la ville.",
    "Il a perdu ses clés dans la cour de l'école ce matin."
];

let pile = [];
let phraseActuelle = "";
let totalEssais = 0;
let totalSucces = 0;

function nouvellePhrase() {
    if (pile.length === 0) pile = [...DICTEES].sort(() => Math.random() - 0.5);
    phraseActuelle = pile.pop();
    document.getElementById('setup-zone').style.display = "none";
    document.getElementById('correction-zone').style.display = "none";
    document.getElementById('dictation-zone').style.display = "block";
    setTimeout(lirePhrase, 500);
}

function lirePhrase() {
    const synthese = window.speechSynthesis;
    const prononciation = new SpeechSynthesisUtterance(phraseActuelle);
    prononciation.lang = 'fr-FR';
    prononciation.rate = 0.8;
    synthese.speak(prononciation);
}

function montrerCorrection() {
    document.getElementById('display-phrase').innerText = phraseActuelle;
    document.getElementById('correction-zone').style.display = "block";
    document.getElementById('dictation-zone').style.display = "none";
}

function enregistrerResultat(estCorrect) {
    totalEssais++;
    if (estCorrect) totalSucces++;
    updateScore();
    nouvellePhrase();
}

function updateScore() {
    const pourcent = Math.round((totalSucces / totalEssais) * 100);
    const bar = document.getElementById('score-bar');
    const txt = document.getElementById('score-text');
    bar.style.width = pourcent + "%";
    if (pourcent < 50) {
        bar.style.backgroundColor = "var(--error)";
        txt.innerHTML = `Précision : ${pourcent}% (Insuffisant)`;
    } else if (pourcent < 80) {
        bar.style.backgroundColor = "var(--warning)";
        txt.innerHTML = `Précision : ${pourcent}% (En progrès)`;
    } else {
        bar.style.backgroundColor = "var(--accent)";
        txt.innerHTML = `Précision : ${pourcent}% (Excellent !)`;
    }
}

// ===== MOTEUR IA =====
function afficherToast(msg, erreur) {
    const t = document.getElementById('toast-ia');
    t.textContent = msg;
    t.style.background = erreur ? '#e74c3c' : '#27ae60';
    t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 3500);
}

async function genererIA() {
    const btn = document.getElementById('btn-ia');
    btn.disabled = true;
    btn.textContent = '⏳ Génération en cours...';
    try {
        const prompt = `Tu es un expert en grammaire française pour le concours CREM au Sénégal.
Génère exactement 10 phrases de dictée de niveau concours, avec des contextes sénégalais/africains.
Chaque phrase doit contenir au moins un piège grammatical : accord du participe passé, homophones, accord sujet-verbe complexe, ou subjonctif.
Retourne UNIQUEMENT un tableau JSON valide de 10 chaînes de caractères, sans texte avant ni après, sans markdown.
Exemple de format : ["Phrase 1 correctement écrite.", "Phrase 2 correctement écrite."]`;

        const response = await fetch('/.netlify/functions/mistral', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                model: 'mistral-small-latest',
                temperature: 0.85,
                messages: [{ role: 'user', content: prompt }]
            })
        });
        const data = await response.json();
        let texte = data.choices[0].message.content.trim().replace(/```json|```/g, '').trim();
        const nouvelles = JSON.parse(texte);
        if (!Array.isArray(nouvelles) || nouvelles.length === 0) throw new Error('Format invalide');
        pile.push(...nouvelles);
        afficherToast('✅ ' + nouvelles.length + ' nouvelles phrases ajoutées !');
    } catch(err) {
        afficherToast('❌ Erreur : ' + err.message, true);
    } finally {
        btn.disabled = false;
        btn.textContent = '🤖 Générer 10 nouvelles phrases avec l\'IA';
    }
}
</script>
